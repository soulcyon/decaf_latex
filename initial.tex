\begin{algorithm}
\caption{SeedSubTrees($\Gamma$)}
\label{alg:seedsubtrees}
\begin{algorithmic}[1]
\FOR{$comp \in \{Components\}$}
   \STATE $level = \{\};$ %\COMMENT{Level of a SubTree}
   \STATE $nFailed = \{0,0,...0\};$ \COMMENT {Counts Failed Components}
   \STATE $\beta = \{$Array of Linked Lists$\};$ \COMMENT {Breadth First History of Trees}
   %\STATE $\Gamma Cache = \{$Array of Size $2 ^{|\Gamma_{comp}|}\};$ \COMMENT {Stores All Possible Subsets of $\Gamma_{comp}$} 
   \IF{(NotEmpty($\Gamma_{comp}$))}
   	\STATE append $comp$ to $level;$
    \STATE $nFailed[comp] = 1;$
    %\STATE $\Gamma Cache_{comp} = \mathcal{P}(\Gamma_{comp});$ \COMMENT{Power Set (Ordered Set) is Cached}
    \STATE append $|$ to $\beta[comp];$
    \STATE GrowSubTrees($level,\, nFailed,\, \beta,\, 1$);
   \ENDIF
\ENDFOR
\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\caption{GrowSubTree($level,\, nFailed,\, \beta,\, subTreeRate$)}
\label{alg:growsubtree}
\begin{algorithmic}[1]
\STATE $FC =$ FailedComponents($level$)$;$
\STATE $\times\mathcal{P}(\Gamma) = \bigtimes\limits_{i=1}^{|FC|}\mathcal{P}(\Gamma_{FC[i]});$ 
\\ \COMMENT{Cartesian Product (Power Sets (Ordered Set))}
  \FOR{$\times\Gamma \in \times\mathcal{P}(\Gamma)$}
    \STATE $nextLevel = \times\Gamma;$ %\COMMENT{Next Level of SubTree}
    \STATE $nextFC =$ FailedComponents($nextLevel$)$;$
    \FOR{$parC \in FC$}
      \FOR{$childC \in nextLevel$}
        \IF{$childC \in nextFC$}
          \IF{$nFailed[childC] == redundancy(childC)$}
            \STATE goto line 3$;$ 
            \COMMENT {Invalid Subtree as it Requires More Components of a Type than in System}   
            \STATE $nFailed[childC] = nFailed[childC] + 1;$
            \STATE append $|$ to $\beta_{childC};$ \COMMENT {Expend a Component}
            \STATE $subTreeRate = subTreeRate * \phi_{parC,\; childC};$ \COMMENT {Update SubTree Rate}
          \ELSE
           \STATE append $parC$ to $\beta_{childC};$ \COMMENT{Mark Could Have Failed, But Did Not Fail}
         \ENDIF
       \ENDIF
      \ENDFOR
    \ENDFOR
    \IF{AtleastOneChildAdded}
      \STATE GrowSubTree($nextLevel,\, nFailed,\, \beta,\, subTreeRate$); \COMMENT {SubTree Can Be Grown Further}
    \ELSE
      \STATE ProcessRates($nFailed,\, \beta,\, subTreeRate$); \COMMENT {SubTree is Stunted}
    \ENDIF
\ENDFOR
\end{algorithmic}
\end{algorithm}
